all: clean BasicShell

BasicShell: main.o arguments.o execute.o history.o builtinfunc.o
        gcc -g -Wall -o BasicShell main.o execute.o history.o builtinfunc.o arguments.o

        main.o: main.c arguments.h execute.h history.h builtinfunc.h
        gcc -g -Wall -c main.c

        arguments.o: arguments.c arguments.h
        gcc -g -Wall -c arguments.c

        execute.o: execute.c execute.h arguments.h builtinfunc.h
        gcc -g -Wall -c execute.c

        history.o: history.c history.h
        gcc -g -Wall -c history.c

        builtinfunc.o: builtinfunc.c builtinfunc.h
        gcc -g -Wall -c builtinfunc.c

        clean:
rm -f BasicShell *.o
